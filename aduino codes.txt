#include <SoftwareSerial.h>
#include <Stepper.h>

const int motorPin1 = 13;
const int motorPin2 = 2;
const int motorPin3 = 7;
const int motorPin4 = 8;

const int stepsPerRevolution = 2048;
Stepper mystepper(stepsPerRevolution, motorPin1, motorPin3, motorPin2, motorPin4);

bool stepperOn = false;

const int INA = 5;
const int INB = 6;
const int INC = 3;
const int IND = 4;
const int ENA = 9;   // PWM 핀
const int ENB = 10;  // PWM 핀

SoftwareSerial BTSerial(11, 12);  // 소프트웨어 시리얼 (TX,RX)

void setup() {
  Serial.begin(9600);
  BTSerial.begin(9600);
  mystepper.setSpeed(14);  // 스텝모터 속도 (RPM)

  pinMode(INA, OUTPUT);
  pinMode(INB, OUTPUT);
  pinMode(INC, OUTPUT);
  pinMode(IND, OUTPUT);
  pinMode(ENA, OUTPUT);
  pinMode(ENB, OUTPUT);
}

void loop() {
  digitalWrite(INA, HIGH);
  digitalWrite(INB, LOW);
  digitalWrite(INC, HIGH);
  digitalWrite(IND, LOW);

  byte data = 0;

  // 블루투스 → 시리얼
  if (BTSerial.available()) {
    data = BTSerial.read();
    Serial.write(data);

    if (data == '0') {
      analogWrite(ENA, 0);
      analogWrite(ENB, 0);
      Serial.println("Speed set to 0");
    } 
    else if (data == '1') {
      analogWrite(ENA, 70);
      analogWrite(ENB, 70);
      Serial.println("Speed set to 100");
    } 
    else if (data == '2') {
      analogWrite(ENA, 75);
      analogWrite(ENB, 75);
      Serial.println("Speed set to 120");
    } 
    else if (data == '3') {
      analogWrite(ENA, 80);
      analogWrite(ENB, 80);
      Serial.println("Speed set to 130");
    } 
    else if (data == '4') {
      analogWrite(ENA, 85);
      analogWrite(ENB, 85);
      Serial.println("Speed set to 140");
    } 
    else if (data == '5') {
      analogWrite(ENA, 100);
      analogWrite(ENB, 100);
      Serial.println("Speed set to 150");
    } 
    else if (data == '6') {
      analogWrite(ENA, 110);
      analogWrite(ENB, 110);
      Serial.println("Speed set to 160");
    } 
    else if (data == '7') {
      analogWrite(ENA, 65);
      analogWrite(ENB, 65);
      Serial.println("Speed set to 170");
    } 
    else if (data == '8') {
      analogWrite(ENA, 0);
      analogWrite(ENB, 140);
      Serial.println("Speed set to 180");
    } 
    else if (data == '9') {
      analogWrite(ENA, 0);
      analogWrite(ENB, 160);
      Serial.println("Speed set to 200");
    } 

  }

  // 시리얼 → 블루투스
  if (Serial.available()) {
    data = Serial.read();
    BTSerial.write(data);
  }

  mystepper.step(1);  // 계속 한 스텝씩 회전
}
